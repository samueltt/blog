冒泡排序
```
#!/usr/bin/env python

def bubblesort(array):
    l = len(array)
    for i in range(l):
        for j in range(l-i):
            if j+1 == l:
                break
            if array[j] > array[j + 1]:
                array[j], array[j + 1] = array[j + 1], array[j]
    return array

array = [1,2,3,6,5,4,6,10]

print bubblesort(array)
```
